Порядок символов при этом должен сохраняться, а замена — быть уникальной. Так, два разных символа строки s нельзя заменить одним и тем же символом из строки t, а вот одинаковые символы в строке s должны заменяться одним и тем же символом.

def is_isomorphic(s, t):
    '''
    s = 'paper'
    t = 'title'
    Алгоритм:
    Находим в строке уникальные символы
    Для каждого уникального символа находим позиции в строке
    Для
        paper = {p: [0, 2], a: [1], e: [3], r: [4]}
        title = {t: [0, 2], i: [1], l: [3], e: [4]}
        apple = {a: [0], p: [1, 2], l: [3], e: [4]}
    Слова изоморфны, если значения словарей совпадают.
    '''
    if len(s) != len(t):
        return False
    dict_s = {}
    for i, ch in enumerate(s):
        if ch not in dict_s.keys():
            dict_s[ch] = []
        dict_s[ch].append(i)

    dict_t = {}
    for i, ch in enumerate(t):
        if ch not in dict_t.keys():
            dict_t[ch] = []
        dict_t[ch].append(i)

    #print("dict_s.values():", dict_s.values())
    #print("dict_t.values():", dict_t.values())

    if tuple(dict_s.values()) == tuple(dict_t.values()):
        return True
    
    return False

def missing_number(nums):
    '''
    Реализовать функцию (или тело функции), которая находит единственное
    отсутствующее число из последовательности натуральных чисел 1,2,…,n.
    '''
    start_n = nums[0]
    for n in nums[1:]:
        if start_n + 1 != n:
            return start_n + 1
        start_n +=1
    return 'Не найдено'

def prime_factors(n):
    '''
    Реализовать функцию (или тело функции), которая при введении натурального 
    числа n разбивает его на простые множители (представить его в виде простых чисел
    ).
    n = 56
    print(prime_factors(n))
    # Вывод:  [2, 2, 2, 7]
    '''
    delit = 2
    result = []
    while n &lt;= delit:
        #print("delit:", delit )
        if n % delit == 0:
            result.append(delit)
            n = n // delit
            #print("n:", n, "delit:", delit )
        else:
            delit +=1

    return result

create table account
(

amount numeric(10,2), -- сумма транзакции
type varchar(3) -- тип транзакции
)

Вывести ID клиентов, которые за последний месяц по всем своим счетам совершили покупок меньше, чем на 5000 рублей. Без использования подзапросов и оконных функций.
<p>
Решение для PostgreSQL:
<pre>
SELECT client_id
FROM account
JOIN transaction t ON account.id = t.account_id -- соединили таблицы
WHERE transaction_date &lt;= (CURRENT_DATE - INTERVAL '1 month')
GROUP BY client_id
HAVING SUM(amount) &lt; 5000;
</pre>
